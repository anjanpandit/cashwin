let localStream;
let remoteStream;
let peerConnection;

const servers = {
  iceServers: [
    { urls: 'stun:stun.l.google.com:19302' }
  ]
};

const startCall = async () => {
  localStream = await navigator.mediaDevices.getUserMedia({ video: true, audio: true });
  document.getElementById('localVideo').srcObject = localStream;

  peerConnection = new RTCPeerConnection(servers);

  localStream.getTracks().forEach(track => {
    peerConnection.addTrack(track, localStream);
  });

  peerConnection.ontrack = event => {
    remoteStream = event.streams[0];
    document.getElementById('remoteVideo').srcObject = remoteStream;
  };

  const offer = await peerConnection.createOffer();
  await peerConnection.setLocalDescription(offer);

  // এই অংশে তোমাকে সিগনালিং সার্ভার যুক্ত করতে হবে future development-এর জন্য
  console.log('Offer created. Implement signaling for real calls.');
};

document.getElementById('startButton').addEventListener('click', startCall);
